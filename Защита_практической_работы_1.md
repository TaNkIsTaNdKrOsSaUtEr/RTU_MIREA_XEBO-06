class Book:
    def __init__(self, title, author, pages, year, price):
        self.title = title
        self.author = author
        self.pages = pages
        self.year = year
        self.price = price

    def describe(self):
        print(f"–ö–Ω–∏–≥–∞ '{self.title}', –∞–≤—Ç–æ—Ä {self.author}, {self.pages} —Å—Ç—Ä.")

    def discount(self, percent):
        self.price *= (1 - percent/100)

    def is_old(self):
        return (2025 - self.year) > 20

    def change_price(self, new_price):
        self.price = new_price

# –°–æ–∑–¥–∞–µ–º –æ–±—ä–µ–∫—Ç—ã
